- Code exact :
  ```php
  namespace App\Services;
  use Illuminate\Support\Facades\Http;
  use Illuminate\Support\Facades\Log;

  class WeatherService
  {
      private $apiKey;
      private $baseUrl = '[https://api.openweathermap.org/data/2.5/weather](https://api.openweathermap.org/data/2.5/weather)';

      public function __construct()
      {
          $this->apiKey = env('OPENWEATHER_API_KEY');
      }

      public function getDailyWeather(string $city)
      {
          if (!$this->apiKey) return $this->getMockedData();

          try {
              $response = Http::timeout(5)->get($this->baseUrl, [
                  'q' => $city . ',MA',
                  'appid' => $this->apiKey,
                  'units' => 'metric',
                  'lang' => 'fr'
              ]);

              if ($response->successful()) {
                  $data = $response->json();
                  return [
                      'temperature' => $data['main']['temp'],
                      'humidite' => $data['main']['humidity'],
                      'pluie_mm' => $data['rain']['1h'] ?? 0,
                  ];
              }
              return $this->getMockedData();
          } catch (\Exception $e) {
              Log::error("Erreur Météo: " . $e->getMessage());
              return $this->getMockedData();
          }
      }

      private function getMockedData() {
          return ['temperature' => rand(20, 35), 'humidite' => rand(40, 80), 'pluie_mm' => rand(0, 5)];
      }

Crée ce service qui croise la plante, le stade de culture et la météo.

Code exact :

PHP

namespace App\Services;
use App\Models\Culture;
use App\Models\Alerte;
use Carbon\Carbon;
use Illuminate\Support\Facades\Log;

class CalculateurService
{
    protected $weatherService;

    public function __construct(WeatherService $weatherService)
    {
        $this->weatherService = $weatherService;
    }

    public function genererRecommandation(Culture $culture)
    {
        try {
            $joursEcoules = Carbon::parse($culture->date_plantation)->diffInDays(now());
            $stade = $culture->plante->stades()->where('jour_debut', '<=', $joursEcoules)->where('jour_fin', '>=', $joursEcoules)->first();

            if (!$stade) return null;

            $meteo = $this->weatherService->getDailyWeather($culture->localisation ?? 'Casablanca');
            $base = ($culture->plante->besoin_eau_base === 'eleve') ? 3 : (($culture->plante->besoin_eau_base === 'moyen') ? 2 : 1);
            $besoinTotal = round($base * $stade->multiplicateur_eau * $culture->superficie);
            $pluie = $meteo['pluie_mm'] ?? 0;

            if ($pluie > 2) {
                $titre = "Arrosage suspendu (Pluie)";
                $message = "Des précipitations ({$pluie}mm) sont prévues à {$culture->localisation}. Inutile d'arroser votre {$culture->plante->nom}.";
                $niveau = 'success';
            } else {
                $titre = "Action requise : Arrosage";
                $message = "Stade: {$stade->nom_stade}. Météo: {$meteo['temperature']}°C. Apportez environ {$besoinTotal}L à votre {$culture->plante->nom}.";
                $niveau = 'warning';
            }

            return Alerte::create([
                'user_id' => $culture->user_id,
                'titre' => $titre,
                'message' => $message,
                'niveau' => $niveau,
                'lue' => false,
            ]);
        } catch (\Exception $e) {
            Log::error("Erreur Calculateur: " . $e->getMessage());
            return null;
        }
    }
}

public function handle(\App\Services\CalculateurService $calculateur)
{
    $cultures = \App\Models\Culture::with('plante')->get(); 
    $count = 0;
    foreach ($cultures as $culture) {
        if ($calculateur->genererRecommandation($culture)) $count++;
    }
    $this->info("Terminé. {$count} alertes générées.");
}
